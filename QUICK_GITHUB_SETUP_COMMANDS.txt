# Quick GitHub Setup Commands for CloudPanel
# Copy and paste these commands one by one into your terminal

# 1. Navigate to project directory
cd /home/fitsmart/htdocs/www.fitsmart.ca

# 2. Fix ownership issue (if running as root)
# Option A: Global (affects all git, but safe - only adds this directory)
git config --global --add safe.directory /home/fitsmart/htdocs/www.fitsmart.ca
# Option B: Local (only affects this project - use this if you prefer isolation)
# git config --local --add safe.directory /home/fitsmart/htdocs/www.fitsmart.ca

# 3. Check if git is initialized
git status

# 4. If not initialized, initialize git
git init

# 5. Configure git user
git config user.name "FitSmart"
git config user.email "122971823+sorru43@users.noreply.github.com"

# 6. Add GitHub remote (or update if exists)
git remote add origin https://github.com/sorru43/fitsmart.ca.git
# OR if remote exists:
# git remote set-url origin https://github.com/sorru43/fitsmart.ca.git

# 7. Verify remote
git remote -v

# 8. For HTTPS authentication, configure credential helper
git config --global credential.helper store

# 9. Rename branch to main (if it was created as master)
git branch -m master main

# 10. Pull from GitHub (will prompt for credentials first time)
git pull origin main --allow-unrelated-histories
# OR if branch is 'master':
# git pull origin master --allow-unrelated-histories

# 11. If you have local changes to commit first:
# git add .
# git commit -m "Initial commit from CloudPanel"

# 12. Push to GitHub (will prompt for credentials first time)
# Username: sorru43
# Password: [your GitHub Personal Access Token]
git push -u origin main
# OR if branch is 'master':
# git push -u origin master
